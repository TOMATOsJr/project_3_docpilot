% ===== DIAGRAM 3: Command + Memento =====
\documentclass[border=8pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta, positioning, calc}
\usepackage{xcolor}
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}

\definecolor{headingblue}{RGB}{31, 73, 125}
\definecolor{accentblue}{RGB}{70, 130, 180}
\definecolor{classblue}{RGB}{210, 225, 245}
\definecolor{interfaceyellow}{RGB}{255, 248, 220}
\definecolor{concretemint}{RGB}{220, 245, 220}

\begin{document}
\begin{tikzpicture}[
  iface/.style={
    rectangle, draw=accentblue, line width=1.2pt,
    fill=interfaceyellow,
    text width=5.0cm, align=center, font=\small
  },
  cls/.style={
    rectangle, draw=headingblue, line width=1.2pt,
    fill=classblue,
    text width=4.8cm, align=center, font=\small
  },
  memcls/.style={
    rectangle, draw=headingblue, line width=1.2pt,
    fill=concretemint,
    text width=4.8cm, align=center, font=\small
  },
  impl/.style={-{Triangle[open,length=9pt,width=7pt]}, line width=1.1pt, dashed, draw=accentblue},
  uses/.style={-{Stealth[length=8pt]}, line width=1.1pt, draw=headingblue},
  assoc/.style={-{Stealth[length=8pt]}, line width=1.1pt, draw=gray!70},
  node distance=1.6cm and 1.8cm
]

%% EditCommand interface (top centre)
\node[iface] (icmd) at (4.5, 0) {%
  \textcolor{accentblue}{\scriptsize\guillemotleft interface\guillemotright}\\[1pt]
  \textbf{EditCommand}\\[1pt]
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize +execute(): void\\
  \scriptsize +undo(): void\\
  \scriptsize +getDescription(): String
};

%% Concrete commands (row below interface)
\node[cls, below=2.2cm of icmd, xshift=-5.4cm] (ins) {%
  \textbf{InsertBlockCommand}\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize -block: DocumentBlock\\
  \scriptsize -position: int\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize +execute(): void\\
  \scriptsize +undo(): void
};
\node[cls, right=0.5cm of ins] (del) {%
  \textbf{DeleteBlockCommand}\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize -block: DocumentBlock\\
  \scriptsize -savedMemento: DocumentMemento\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize +execute(): void\\
  \scriptsize +undo(): void
};
\node[cls, right=0.5cm of del] (rep) {%
  \textbf{ReplaceBlockCommand}\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize -oldBlock: DocumentBlock\\
  \scriptsize -newBlock: DocumentBlock\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize +execute(): void\\
  \scriptsize +undo(): void
};

%% DocumentEditor (Originator) — right side
\node[cls, right=3.2cm of icmd] (editor) {%
  \textbf{DocumentEditor}\\
  {\scriptsize\textit{(Originator)}}\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize -blockTree: BlockTree\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize +applyCommand(cmd: EditCommand)\\
  \scriptsize +createMemento(): DocumentMemento\\
  \scriptsize +restoreMemento(m)
};

%% DocumentMemento — below editor
\node[memcls, below=2.2cm of editor] (memento) {%
  \textbf{DocumentMemento}\\
  {\scriptsize\textit{(Memento)}}\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize -snapshot: BlockTree\\
  \scriptsize -timestamp: DateTime\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize +getSnapshot(): BlockTree
};

%% EditHistoryManager (Caretaker) — far right
\node[cls, right=2.0cm of editor] (hist) {%
  \textbf{EditHistoryManager}\\
  {\scriptsize\textit{(Caretaker)}}\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize -undoStack: Stack$\langle$EditCommand$\rangle$\\
  \scriptsize -redoStack: Stack$\langle$EditCommand$\rangle$\\
  \rule{\linewidth}{0.4pt}\\[1pt]
  \scriptsize +executeCommand(cmd)\\
  \scriptsize +undo()\\
  \scriptsize +redo()
};

%% Arrows
% Concrete commands implement EditCommand
\draw[impl] (ins.north)  -- ++(0,0.4) -| (icmd.south west);
\draw[impl] (del.north)  -- (icmd.south);
\draw[impl] (rep.north)  -- ++(0,0.4) -| (icmd.south east);

% Editor creates Memento
\draw[uses] (editor.south) -- node[right, font=\scriptsize]{creates} (memento.north);

% Caretaker manages Editor
\draw[assoc] (hist.west) -- node[above, font=\scriptsize]{manages} (editor.east);

% Caretaker stores mementos
\draw[assoc, bend right=25] (hist.south) to node[right, font=\scriptsize, pos=0.4]{stores} (memento.east);

% DeleteBlockCommand saves/restores memento
\draw[assoc, dashed, draw=gray!60, bend left=15]
  (del.east) to node[below, font=\scriptsize]{saves/restores} (memento.west);

\end{tikzpicture}
\end{document}
